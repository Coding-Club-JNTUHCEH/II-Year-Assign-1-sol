#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cout<<"Enter no of food items"<<endl;
	cin>>n;
	int req=(n/2);
	req+=(n%2);
	int items;
	int count=0;
	cout<<"enter delivered times of n food items"<<endl;
	for(int i=0;i<n;i++)
	{
		cin>>items;
		if(items<=30)
		count++;
	}
	if(count>=req)
	{
		cout<<"Satisfied"<<endl;
	}
	else
	{
		cout<<"Frustated"<<endl;
	}
}
